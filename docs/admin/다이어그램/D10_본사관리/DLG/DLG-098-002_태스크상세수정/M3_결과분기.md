# M3 결과 분기 플로우 — DLG-098-002 태스크 상세/수정

```mermaid
flowchart TD
  Actions([수정 / 삭제 / 완료 처리]) --> ActionType{액션 유형}

  ActionType -->|": 수정"|SubmitEdit[동작]
  ActionType -->|": 삭제"|SubmitDel[동작]
  ActionType -->|": 완료 체크"|SubmitComplete[동작]

  SubmitEdit --> R_Edit{응답}
  R_Edit -->|": 200"|EditOK[닫힘 + 목록 갱신]
  R_Edit -->|": 400"|EditErr[인라인 에러]
  R_Edit -->|": 500"|EditFail[toast\n"저장에 실패했습니다."]
  EditOK --> Toast_Edit((toast\n"태스크가 수정되었습니다."))

  SubmitDel --> R_Del{응답}
  R_Del -->|": 200"|DelOK[닫힘 + 목록 제거]
  R_Del -->|": 500"|DelFail[toast\n"삭제에 실패했습니다."]
  DelOK --> Toast_Del((toast\n"태스크가 삭제되었습니다."))

  SubmitComplete --> R_Complete{응답}
  R_Complete -->|": 200"|CompleteOK[완료 상태 + 취소선]
  R_Complete -->|": 500"|CompleteFail[toast\n"상태 변경에 실패했습니다."]
  CompleteOK --> AllCheck{전체 완료?}
  AllCheck -->|": Yes"|AllDone((toast\n"오늘 할일을 모두 완료했습니다!"))
  AllCheck -->|": No"|Toast_Complete((toast\n"완료 처리되었습니다."))

  classDef system fill:#EDE7F6,stroke:#5E35B2
  classDef success fill:#E8F5E9,stroke:#2E7D32,color:#1B5E20
  classDef error fill:#FFEBEE,stroke:#C62828,color:#B71C1C

  class Actions,ActionType,SubmitEdit,SubmitDel,SubmitComplete,R_Edit,R_Del,R_Complete,AllCheck system
  class EditOK,DelOK,CompleteOK,Toast_Edit,Toast_Del,Toast_Complete,AllDone success
  class EditErr,EditFail,DelFail,CompleteFail error
```
