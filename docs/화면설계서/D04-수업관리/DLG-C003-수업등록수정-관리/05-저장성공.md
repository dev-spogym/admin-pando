```
const handleSubmit = async (data: FormValues) => {
  setIsSubmitting(true)
  try {
    if (mode === 'create') {
      await fetch('/api/lessons', { method: 'POST', body: JSON.stringify(data) })
      toast.success("수업이 등록되었습니다.")
    } else {
      await fetch(`/api/lessons/${lesson!.id}`, { method: 'PUT', body: JSON.stringify(data) })
      toast.success("수업이 수정되었습니다.")
    }
    onSuccess()  // 목록 리프레시 + 모달 닫힘
  } catch (error) {
    handleApiError(error)
  } finally {
    setIsSubmitting(false)
  }
}

저장 중 상태 (isSubmitting=true):
- 저장 버튼: disabled + Loader2 스피너
- 폼 필드: pointer-events-none opacity-50

API 에러 분기 (handleApiError):
- 409: toast.warning "동일한 수업명이 존재합니다."
- 403: toast.warning "권한이 없습니다."
- 500: toast.error "처리에 실패했습니다."
```
