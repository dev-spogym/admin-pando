## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/AutoPenaltyPolicyModal.tsx (handleSubmit)

const handleSubmit = async (data: PolicyValues) => {
  setIsSubmitting(true)
  try {
    await fetch('/api/penalty-policy', {
      method: 'PUT',
      body: JSON.stringify(data),
    })
    // A05 크론 정책 갱신 (서버 사이드에서 자동 처리)
    queryClient.invalidateQueries({ queryKey: ['penalty-policy'] })
    toast.success("정책이 저장되었습니다.")
    onClose()
  } catch (error) {
    if (error.status === 400) toast.warning("입력값을 확인해주세요.")
    else if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsSubmitting(false)
  }
}

저장 중 UI: 저장 버튼 disabled + Loader2
```
