## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/SessionDetailModal.tsx (내부 confirm 다이얼로그)

차감 확인 다이얼로그 (showDeductConfirm=true):
<div className="absolute inset-0 bg-black/30 flex items-center justify-center rounded-xl">
  <div className="bg-white rounded-lg p-6 w-72 shadow-xl">
    <h3 className="font-semibold">횟수를 차감하시겠습니까?</h3>
    <p className="mt-2 text-sm text-gray-500">
      잔여 {session.remaining_count}회 → {session.remaining_count - 1}회
    </p>
    <div className="mt-4 flex justify-end gap-3">
      <Button variant="outline" onClick={() => setShowDeductConfirm(false)}>취소</Button>
      <Button onClick={handleDeduct} disabled={isDeducting}>
        {isDeducting ? <Loader2 className="animate-spin w-4 h-4" /> : '확인'}
      </Button>
    </div>
  </div>
</div>

handleDeduct:
const handleDeduct = async () => {
  setIsDeducting(true)
  try {
    await fetch(`/api/sessions/${sessionId}/deduct`, { method: 'POST' })
    toast.success("횟수가 차감되었습니다.")
    refetch()  // 잔여횟수 갱신
    setShowDeductConfirm(false)
    onSuccess()
  } catch (error) {
    if (error.status === 400) toast.warning("잔여 횟수가 없습니다.")
    else if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsDeducting(false)
  }
}
```
