- 정렬 기본 `regDate DESC`, 컬럼 헤더 토글 가능
- 검색 trim + 소문자 비교
- `/branches/:id/stats` 병렬 호출 캐시
- windowFocus 자동 refetch
- 엑셀 필드: id,name,code,phone,address,status,members,staffs,monthlyRevenue,regDate
- audit_log: switchBranch → BRANCH_SWITCH, export → EXPORT

## 에지 케이스
- 검색 결과 0건: `03-빈목록` 서브상태

- 상태 변경 optimistic update → 실패 시 rollback + toast.error

## 바이브코딩 프롬프트 (델타)
```
async function switchBranch(id, name) {
  await useBranchStore.getState().setCurrent({ id, name });
  await api.post(`/branches/${id}/switch`);
  toast.success(`${name} 지점으로 전환되었습니다`);
  router.push('/');
}

async function handleStatusChange(row, newStatus) {
  if (newStatus === 'closed') {
    const impact = await api.get(`/branches/${row.id}/close-impact`);
    if (impact.activeMembers > 0 || impact.activeStaffs > 0) {
      setStatusChangeDialog({ row, newStatus, impact });
      return;
    }
  }
  setStatusChangeDialog({ row, newStatus, impact: null });
}

const confirmStatusChange = async () => {
  try {
    await api.patch(`/branches/${row.id}/status`, { status, reason });
    queryClient.invalidateQueries(['branches']);
    toast.success(`"${row.name}" 지점이 ${STATUS_LABEL[status]} 처리되었습니다.`);
  } catch (e) {
    if (e.errorCode === 'E422900') setDialogError('이관/환불 미처리 회원이 있어 폐점할 수 없습니다.');
    else toast.error(`상태 변경 실패: ${e.message}`);
  }
};

const handleExport = () => {
  exportToExcel(filteredBranches, exportColumns, { filename:'지점목록' });
  api.post('/audit-log', { action:'EXPORT', targetType:'branch' });
  toast.success(`${filteredBranches.length}건 엑셀 다운로드 완료`);
};

// debounce 검색
const debouncedKeyword = useDebounce(keyword, 300);
const filtered = useMemo(() => branches.filter(b =>
  (!debouncedKeyword || b.name.includes(debouncedKeyword) || b.address.includes(debouncedKeyword)) &&
  (!statusFilter || b.status === statusFilter)
), [branches, debouncedKeyword, statusFilter]);
```
