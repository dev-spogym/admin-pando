## 🧩 바이브코딩 프롬프트

```
모달: DLG-084-001 플랜 변경 확인

<Modal open={showPlanChangeModal}>
  <ModalContent className="max-w-lg">
    <ModalHeader><ModalTitle>플랜 변경</ModalTitle></ModalHeader>
    <ModalBody className="space-y-4">
      <p className="text-sm text-gray-600">
        <strong>{currentPlan.name}</strong> → <strong>{newPlan.name}</strong> 플랜으로 변경합니다.
        다음 결제일부터 새 플랜이 적용됩니다.
      </p>
      <!-- 플랜 비교표 -->
      <div className="grid grid-cols-2 gap-3">
        {[currentPlan, newPlan].map((plan, i) => (
          <div key={i} className={cn(
            "p-4 rounded-lg border",
            i === 1 ? "border-primary bg-primary/5" : "border-gray-200 bg-gray-50"
          )}>
            <p className="text-xs text-gray-400">{i === 0 ? '현재' : '변경 후'}</p>
            <p className="font-bold text-gray-900 mt-0.5">{plan.name}</p>
            <p className="text-sm text-gray-500 mt-1">{plan.price.toLocaleString()}원/월</p>
            <ul className="mt-2 space-y-1 text-xs text-gray-500">
              <li>지점 {plan.max_branches}개</li>
              <li>직원 {plan.max_staff}명</li>
              <li>회원 {plan.max_members}명</li>
            </ul>
          </div>
        ))}
      </div>
    </ModalBody>
    <ModalFooter className="flex justify-end gap-2">
      <Button variant="outline" onClick={() => setShowPlanChangeModal(false)}>취소</Button>
      <Button onClick={handlePlanChange}>변경 확인</Button>
    </ModalFooter>
  </ModalContent>
</Modal>

handlePlanChange: async () => {
  setIsChanging(true)
  const { error } = await fetch('/api/subscription/change', {
    method: 'POST', body: JSON.stringify({ newPlanId: newPlan.id })
  })
  if (error?.status === 402) { toast.error('결제 실패. 카드 정보를 확인해 주세요.'); setIsChanging(false); return }
  if (error) { toast.error('플랜 변경 실패'); setIsChanging(false); return }
  onPlanChanged(newPlan)
  setShowPlanChangeModal(false)
  toast.success('플랜이 변경되었습니다. 다음 결제일부터 적용됩니다.')
}
```
