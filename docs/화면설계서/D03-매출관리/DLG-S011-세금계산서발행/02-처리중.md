```
발행 로직:
const handleIssue = async () => {
  setIsIssuing(true);
  const { error } = await supabase
    .from('tax_invoices')
    .update({
      status: 'ISSUED',
      issued_at: issueDate,
    })
    .eq('id', taxInvoice.id);

  if (error) {
    toast.error('발행에 실패했습니다.');
    setIsIssuing(false);
  } else {
    setShowIssueModal(false);
    await fetchTaxInvoiceDetail(taxInvoice.id);
    toast.success('세금계산서가 발행되었습니다.');
  }
};
```
