  const validate = () => {
    const errs: Record<string, string> = {}
    if (!form.name.trim()) errs.name = '규칙명을 입력해 주세요.'
    if (!form.value || Number(form.value) <= 0) errs.value = '할인 값을 입력해 주세요.'
    if (form.discountType === 'rate' && Number(form.value) > 100) errs.value = '정률 할인은 100% 이하여야 합니다.'
    if (form.condition === 'period' && (!form.start_date || !form.end_date)) errs.period = '적용 기간을 선택해 주세요.'
    return errs
  }

에러 표시:
