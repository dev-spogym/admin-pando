const onSubmit = async (data: TargetFormData) => {
  setIsSubmitting(true)
  try {
    const storageKey = `kpi_monthly_target_${branchId}_${data.year}${String(data.month).padStart(2,'0')}`

    // 기존 목표 덮어쓰기 (E_D94_01_M3_06)
    localStorage.setItem(storageKey, JSON.stringify({
      amount: Number(data.amount),
      memo: data.memo,
      savedAt: new Date().toISOString()
    }))

    // M3 성공 처리
    toast.success("목표가 저장되었습니다.") // Toast_OK
    onClose()        // E_D94_01_M3_03: 닫힘
    onKpiRefresh()   // E_D94_01_M3_04: KPI 카드 목표 달성률 재계산

  } catch (err) {
    if (err.name === 'QuotaExceededError') {
      // E_D94_01_M3_02: localStorage 용량 초과
      // 오래된 항목 정리 후 재시도 (E_D94_01_M3_05)
      cleanOldTargets()
      localStorage.setItem(storageKey, JSON.stringify(data))
      toast.success("목표가 저장되었습니다.")
      onClose()
    } else {
      toast.error("저장에 실패했습니다.")
    }
  } finally {
    setIsSubmitting(false)
  }
}
