```
저장 성공 처리:
const onSubmit = async (data: BranchFormData) => {
  setIsSubmitting(true)
  try {
    const { error } = await supabase.from('branches').insert(data)
    if (error) throw error

    // M3 결과 분기: 201 Created
    toast.success("지점이 등록되었습니다.") // E_D92_01_M3_07
    onClose()       // E_D92_01_M1_05: 다이얼로그 닫힘
    onRefresh()     // E_D92_01_M3_06: SCR-092 목록 갱신

  } catch (err) {
    // 에러 분기 처리 (M3 기반)
    if (err.code === '23505') { // 409 Conflict (지점코드 중복)
      setError('branch_code', { message: '이미 사용 중인 지점코드입니다.' })
    } else if (err.status === 403) {
      toast.error("권한이 없습니다.") // E_D92_01_M3_09
      onClose()
    } else {
      toast.error("저장에 실패했습니다. 다시 시도하세요.") // E_D92_01_M3_05
    }
  } finally {
    setIsSubmitting(false)
  }
}

[저장] 버튼 로딩:
<Button disabled={isSubmitting} onClick={handleSubmit(onSubmit)}>
  {isSubmitting ? <><Loader2 className="w-4 h-4 animate-spin mr-2" />저장 중...</> : '저장'}
</Button>
```
