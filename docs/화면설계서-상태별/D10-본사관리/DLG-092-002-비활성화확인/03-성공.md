```
const handleDeactivate = async () => {
  setIsSubmitting(true)
  try {
    const { error } = await supabase
      .from('branches')
      .update({ status: 'inactive' })
      .eq('id', branchId)

    if (error) throw error

    // M3 결과 분기
    toast.success("지점이 비활성화되었습니다.") // E_D92_02_M3_06 + Toast_OK
    onClose()    // 다이얼로그 닫힘
    onRefresh()  // SCR-092 목록 status → inactive 반영

  } catch (err) {
    if (err.code === '23503' || err.status === 409) {
      // E_D92_02_M3_04: 409 — 활성 회원/직원 존재
      setBlockError(`활성 회원 ${activeMembers}명, 직원 ${activeStaff}명이 있습니다. 먼저 이전하세요.`)
    } else if (err.status === 403) {
      toast.error("권한이 없습니다.") // E_D92_02_M3_08
      onClose()
    } else {
      toast.error("비활성화에 실패했습니다.") // 500
    }
  } finally {
    setIsSubmitting(false)
  }
}
```
