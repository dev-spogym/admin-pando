화면: DLG-S014 환불상세결과 — 상태: 열림 (신규 기획)

<Modal isOpen={isOpen} onClose={onClose} title="환불 상세">
  <div className="p-6 space-y-3">
    <InfoRow label="회원명" value={refund.member_name} />
    <InfoRow label="상품명" value={refund.product_name} />
    <InfoRow label="원 결제금액" value={formatKRW(refund.original_amount)} />
    <InfoRow label="환불 유형"
      value={refund.refund_type === 'full' ? '전액 환불' : '부분 환불'} />
    <InfoRow label="환불 금액"
      value={formatKRW(refund.refund_amount)}
      valueClassName="font-bold text-red-600" />
    <InfoRow label="환불 사유" value={refund.reason} />
    <InfoRow label="처리일시" value={refund.processed_at
      ? new Date(refund.processed_at).toLocaleString('ko-KR')
      : '-'} />
    <InfoRow label="처리자" value={refund.processed_by_name ?? '-'} />
    <InfoRow label="상태"
      value={<StatusBadge
        label={REFUND_STATUS_KO[refund.status]}
        variant={REFUND_STATUS_VARIANT[refund.status]}
      />} />
  </div>

  <div className="p-4 border-t">
    <Button variant="outline" className="w-full" onClick={onClose}>닫기</Button>
  </div>

const REFUND_STATUS_KO = {
  PENDING: '대기중',
  APPROVED: '승인완료',
  REJECTED: '반려',
};
const REFUND_STATUS_VARIANT = {
  PENDING: 'warning',
  APPROVED: 'success',
  REJECTED: 'error',
};

타입:
interface RefundDetail {
  id: number;
  member_name: string;
  product_name: string;
  original_amount: number;
  refund_type: 'full' | 'partial';
  refund_amount: number;
  reason: string;
  processed_at: string | null;
  processed_by_name: string | null;
  status: 'PENDING' | 'APPROVED' | 'REJECTED';
}
