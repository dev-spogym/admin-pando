const fetchRefundDetail = async (id: number) => {
  setIsLoading(true);
  const { data, error } = await supabase
    .from('refunds')
    .select('*')
    .eq('id', id)
    .single();
  if (error) {
    toast.error('데이터를 불러올 수 없습니다.');
    onClose();
  } else {
    setRefund(data);
  }
  setIsLoading(false);
};
