```
  </Button>

저장 로직 (upsert):
const handleSave = async () => {
  setIsProcessing(true);
  const { error } = await supabase
    .from('sales_targets')
    .upsert({
      branch_id: branchId,
      year: targetYear,
      month: targetMonth,
      amount: targetAmount,
    }, { onConflict: 'branch_id,year,month' });

  if (error) {
    toast.error('목표 저장에 실패했습니다.');
    setIsProcessing(false);
  } else {
    onClose();
    onRefresh?.();
    toast.success('목표 매출이 저장되었습니다.');
  }
};
```
