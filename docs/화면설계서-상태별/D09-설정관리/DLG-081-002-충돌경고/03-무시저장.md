## 🧩 바이브코딩 프롬프트

```
handleForceSave: async () => {
  // M3_081002_01: localStorage 저장 (충돌 인정)
  const { error } = await supabase.from('user_permissions').upsert(buildChangeset(permMatrix, savedMatrix))
  if (error) { toast.error('저장 실패'); return }
  setSavedMatrix(deepClone(permMatrix))
  setChangedCells(new Set())
  setIsDirty(false)
  setShowConflictModal(false)
  // M3_081002_04: 인라인 "저장되었습니다"
  setInlineMsg('저장되었습니다.')
  setTimeout(() => setInlineMsg(''), 3000)
}
```
