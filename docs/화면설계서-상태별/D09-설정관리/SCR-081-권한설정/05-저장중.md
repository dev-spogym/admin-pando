```
handleSave: async () => {
  setIsSaving(true)
  
  // 충돌 검사: 삭제 on + 쓰기 off
  const conflicts = detectConflicts(permMatrix)
  if (conflicts.length > 0) {
    setShowConflictModal(true) // DLG-081-002
    setIsSaving(false)
    return
  }
  
  // PATCH /permissions (변경된 항목만)
  const changes = buildChangeset(permMatrix, savedMatrix)
  const { error } = await supabase
    .from('user_permissions')
    .upsert(changes.map(c => ({ branch_id: branchId, ...c })))
  
  if (error) {
    toast.error('권한 저장에 실패했습니다.')
    setIsSaving(false)
    return
  }
  
  setSavedMatrix(deepClone(permMatrix))
  setChangedCells(new Set())
  setIsDirty(false)
  setIsSaving(false)
  // → SavedState: 변경 셀 하이라이트 해제 (E_F6_081_08)
}

저장중 UI:

- 매트릭스 전체: pointer-events-none opacity-60
- 초기화 버튼: disabled
```
