  setIsSaving(true)
  const { error } = await supabase
    .from('kiosk_settings')
    .upsert({ branch_id: branchId, ...form })
  if (error) {
    toast.error('키오스크 설정 저장에 실패했습니다.')
    setIsSaving(false)
    return
  }
  setSavedData({...form})
  setIsDirty(false)
  toast.success('키오스크 설정이 저장되었습니다.')
  setIsSaving(false)
}
