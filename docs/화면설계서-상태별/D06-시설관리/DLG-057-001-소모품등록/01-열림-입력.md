```
모달: DLG-057-001 소모품 등록 (신규 기능)

<Modal isOpen={isOpen} onClose={onClose} title="소모품 등록" size="md">
  <div className="space-y-4">
    {/* 품목명 */}
    <div>
      <label className="text-sm font-medium text-gray-700 mb-1 block">품목명 *</label>
      <input
        type="text"
        className={`w-full border rounded-md p-2 text-sm ${errors.name ? 'border-red-300' : 'border-gray-300'}`}
        placeholder="예: 청소용 락스, 물티슈"
        value={name}
        onChange={e => setName(e.target.value)}
      />
      {errors.name && <p className="text-xs text-red-500 mt-1">{errors.name}</p>}
    </div>

    {/* 카테고리 */}
    <div>
      <label className="text-sm font-medium text-gray-700 mb-1 block">카테고리 *</label>
      <Select
        options={[
          { value: 'cleaning', label: '청소용품' },
          { value: 'hygiene', label: '위생용품' },
          { value: 'sports', label: '운동용품' },
          { value: 'etc', label: '기타' },
        ]}
        value={category}
        onChange={setCategory}
        placeholder="카테고리 선택"
      />
      {errors.category && <p className="text-xs text-red-500 mt-1">{errors.category}</p>}
    </div>

    {/* 단위 */}
    <div>
      <label className="text-sm font-medium text-gray-700 mb-1 block">단위</label>
      <input
        type="text"
        className="w-full border border-gray-300 rounded-md p-2 text-sm"
        placeholder="예: 개, 박스, L, kg"
        value={unit}
        onChange={e => setUnit(e.target.value)}
      />
    </div>

    {/* 초기 재고 */}
    <div>
      <label className="text-sm font-medium text-gray-700 mb-1 block">초기 재고</label>
      <input
        type="number"
        className="w-full border border-gray-300 rounded-md p-2 text-sm"
        placeholder="0"
        min={0}
        value={currentStock}
        onChange={e => setCurrentStock(Number(e.target.value))}
      />
    </div>

    {/* 안전재고 */}
    <div>
      <label className="text-sm font-medium text-gray-700 mb-1 block">안전재고 *</label>
      <input
        type="number"
        className={`w-full border rounded-md p-2 text-sm ${errors.safetyStock ? 'border-red-300' : 'border-gray-300'}`}
        placeholder="0"
        min={0}
        value={safetyStock}
        onChange={e => setSafetyStock(Number(e.target.value))}
      />
      {errors.safetyStock && <p className="text-xs text-red-500 mt-1">{errors.safetyStock}</p>}
      <p className="text-xs text-gray-400 mt-1">현재고가 안전재고 미만 시 자동 발주 알림</p>
    </div>
  </div>

  <div className="flex justify-end gap-2 mt-6">
    <Button variant="outline" onClick={onClose}>취소</Button>
    <Button variant="primary" onClick={handleSubmit} loading={isSubmitting}>저장</Button>
  </div>
</Modal>
```
