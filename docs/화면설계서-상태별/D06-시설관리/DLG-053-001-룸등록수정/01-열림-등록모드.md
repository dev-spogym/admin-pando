        onChange={setType}
        placeholder="유형 선택"
      />
      {errors.type && <p className="text-xs text-red-500 mt-1">{errors.type}</p>}

        min={1}

        onChange={e => setCapacity(Number(e.target.value))}
      />
    </div>

    {/* 초기 상태 */}
    <div>
      <label className="text-sm font-medium text-gray-700 mb-1 block">초기 상태</label>
