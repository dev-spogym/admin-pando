## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/PenaltyRegisterModal.tsx (handleSubmit)

const handleSubmit = async (data: PenaltyValues) => {
  setIsSubmitting(true)
  try {
    await fetch('/api/penalties', {
      method: 'POST',
      body: JSON.stringify(data),
    })
    toast.success("페널티가 등록되었습니다.")
    onSuccess()
    onClose()
  } catch (error) {
    if (error.status === 400) toast.warning("입력값을 확인해주세요.")
    else if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsSubmitting(false)
  }
}

저장 중 UI: 등록 버튼 disabled + Loader2 스피너
```
