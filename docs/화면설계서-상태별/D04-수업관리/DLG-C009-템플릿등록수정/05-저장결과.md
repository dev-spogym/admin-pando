## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/ClassTemplateModal.tsx (handleSubmit)

const handleSubmit = async (data: TemplateValues) => {
  setIsSubmitting(true)
  try {
    if (mode === 'create') {
      await fetch('/api/lesson-templates', { method: 'POST', body: JSON.stringify(data) })
      toast.success("템플릿이 저장되었습니다.")
    } else {
      await fetch(`/api/lesson-templates/${template!.id}`, { method: 'PUT', body: JSON.stringify(data) })
      toast.success("템플릿이 수정되었습니다.")
    }
    onSuccess()
    onClose()
  } catch (error) {
    if (error.status === 409) toast.warning("동일한 템플릿명이 존재합니다.")
    else if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsSubmitting(false)
  }
}
```
