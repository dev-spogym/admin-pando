## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/SessionAdjustModal.tsx (handleSubmit)

const handleSubmit = async (data: AdjustValues) => {
  setIsSubmitting(true)
  try {
    await fetch(`/api/sessions/${sessionId}/adjust`, {
      method: 'POST',
      body: JSON.stringify({ amount: data.amount, reason: data.reason }),
    })
    toast.success("횟수가 조정되었습니다.")
    onSuccess()
    onClose()
  } catch (error) {
    if (error.status === 400) toast.warning("입력값을 확인해주세요.")
    else if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsSubmitting(false)
  }
}

저장 중 UI: 저장 버튼 disabled + Loader2
```
