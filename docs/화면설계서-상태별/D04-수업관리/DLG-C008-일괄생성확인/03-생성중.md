## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/BulkCreateConfirmModal.tsx (handleConfirm)

생성 중 UI (isCreating=true):
- 확인 버튼: disabled + Loader2 스피너
- 취소 버튼: disabled
- 요약 영역: opacity-50 pointer-events-none

const handleConfirm = async () => {
  setIsCreating(true)
  try {
    const result = await fetch('/api/lesson-schedules/bulk', {
      method: 'POST',
      body: JSON.stringify(summary),
    }).then(r => r.json())

    if (result.status === 207) {
      // 부분 성공
      setConflicts(result.conflicts)
    } else {
      toast.success(`${result.createdCount}개 수업이 생성되었습니다.`)
      onSuccess()  // SCR-C003 폼 리셋
      onClose()
    }
  } catch (error) {
    if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsCreating(false)
  }
}
```
