## 🧩 바이브코딩 프롬프트

```
파일: src/components/classes/AlternativeScheduleModal.tsx (handleSubmit)

const handleSubmit = async (data: AltValues) => {
  setIsSubmitting(true)
  try {
    await fetch(`/api/schedule-requests/${requestId}/alternative`, {
      method: 'POST',
      body: JSON.stringify(data),
    })
    toast.success("대안 일정이 전송되었습니다.")
    onSuccess()
    onClose()
  } catch (error) {
    if (error.status === 400) toast.warning("입력값을 확인해주세요.")
    else if (error.status === 404) {
      toast.warning("요청을 찾을 수 없습니다.")
      onClose()  // 이미 처리된 요청 → 닫힘
    }
    else if (error.status === 403) toast.warning("권한이 없습니다.")
    else toast.error("처리에 실패했습니다.")
  } finally {
    setIsSubmitting(false)
  }
}

전송 중 UI: 전송 버튼 disabled + Loader2
```
