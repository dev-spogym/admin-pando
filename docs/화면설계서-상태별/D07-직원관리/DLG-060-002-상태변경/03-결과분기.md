## 🧩 바이브코딩 프롬프트

```
API 호출 핸들러:
  const handleStatusChange = async () => {
    if (!validateStatusChange()) return
    setIsProcessing(true)

    try {
      const { error } = await supabase
        .from('staff')
        .update({
          staff_status: selectedStatus,
          status_reason: statusReason || null,
          status_changed_at: new Date().toISOString(),
          status_changed_by: user.id,
        })
        .in('id', targetStaffIds)

      if (error) throw error

      toast.success('상태가 변경되었습니다.')
      setStatusModalOpen(false)
      setSelectedRows(new Set())
      fetchStaffList()

    } catch (err: any) {
      if (err.code === '42501') {
        toast.error('상태 변경 권한이 없습니다.')
        setStatusModalOpen(false)
      } else if (err.status === 400) {
        toast.error('입력 오류가 발생했습니다.')
        // 모달 유지
      } else {
        toast.error('서버 오류가 발생했습니다. 다시 시도해주세요.')
        // 모달 유지
      }
    } finally {
      setIsProcessing(false)
    }
  }

성공 후 처리:
  1. toast.success('상태가 변경되었습니다.')
  2. setStatusModalOpen(false)
  3. setSelectedRows(new Set())
  4. fetchStaffList() — SCR-060 갱신
```
