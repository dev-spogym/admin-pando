저장 핸들러:
  const handleSavePayroll = async (data: PayrollFormValues) => {
    setIsSaving(true)
    try {
      const { error } = await supabase
        .from('payroll')
        .update({
          base_salary: data.base_salary,
          allowances: data.allowances,
          deductions: data.deductions,
          net_salary: data.base_salary + data.allowances - data.deductions,
          updated_at: new Date().toISOString(),
          updated_by: user.id,
        })
        .eq('id', selectedPayrollId)

      if (error) throw error

      toast.success('급여 정보가 저장되었습니다.')
      setDetailModalOpen(false)
      fetchPayrollList()  // SCR-064 목록 갱신

    } catch (err: any) {
      if (err.code === '42501') {
        toast.error('급여 수정 권한이 없습니다.')
        setDetailModalOpen(false)
      } else if (err.status === 409) {
        toast.error('이미 확정된 급여는 수정할 수 없습니다.')
        setDetailModalOpen(false)
      } else if (err.status === 400) {
        toast.error('입력 값을 확인하세요.')
        // 모달 유지
      } else {
        toast.error('저장에 실패했습니다. 다시 시도해주세요.')
        // 모달 유지, 재시도 가능
      }
    } finally {
      setIsSaving(false)
    }
  }

## 📝 디스크립션

### API 분기 결과
| 응답 | 처리 |
|------|------|
| 200 OK | toast.success + 닫힘 + SCR-064 갱신 |
| 400 | toast.error('입력 값 확인') + 모달 유지 |
| 403/RLS | toast.error('권한 없음') + 닫힘 |
| 409 확정됨 | toast.error('이미 확정된 급여') + 닫힘 |
| 500 | toast.error('서버 오류') + 모달 유지 |

### 비즈니스 룰
- net_salary 서버 재계산: base_salary + allowances - deductions
- 확정 후 수정 시도(409): 클라이언트에서도 disabled로 방지했지만 서버 이중 검증
- 저장 성공 후 SCR-064 통계 카드 자동 갱신
