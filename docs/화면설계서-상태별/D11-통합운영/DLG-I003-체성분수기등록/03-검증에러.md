검증 로직 (M2 플로우 기준):
const handleSubmit = async () => {
  const newErrors: Record<string, string> = {}

  // 1. 회원 선택
  if (!selectedMember && !prefillMemberId) {
    newErrors.member = '회원을 선택해 주세요'
  }

  // 2. 측정일 유효성
  const measureDateObj = new Date(measureDate)
  if (measureDateObj > new Date()) {
    newErrors.date = '미래 날짜는 입력할 수 없습니다'
  }

  // 3. 체중 필수
  if (!formValues.weight) {
    newErrors.weight = '체중을 입력해 주세요'
  } else if (Number(formValues.weight) <= 0 || Number(formValues.weight) > 500) {
    newErrors.weight = '올바른 체중을 입력해 주세요 (0.1 ~ 500kg)'
  }

  // 4. 체지방률 범위 (선택 입력 시)
  if (formValues.bodyFatPercentage &&
      (Number(formValues.bodyFatPercentage) < 0 || Number(formValues.bodyFatPercentage) > 100)) {
    newErrors.bodyFatPercentage = '체지방률은 0~100% 사이여야 합니다'
  }

  // 5. 중복 날짜 검사
  if (!newErrors.date && selectedMemberId) {
    const { data: existing } = await supabase
      .from('bodyComposition')
      .select('id')
      .eq('member_id', selectedMemberId)
      .eq('measured_at', measureDate)
      .single()
    if (existing) {
      setDuplicateDate(true)  // 덮어쓰기 확인 모달
      return
    }
  }

  if (Object.keys(newErrors).length > 0) {
    setErrors(newErrors)
    return
  }
  await submitBodyComp()
}

## 📝 디스크립션

### 검증 순서 (M2 플로우)
1. 회원 미선택
2. 미래 날짜
3. 체중 누락/범위 초과
4. 체지방률 범위 (선택 입력 시)
5. 중복 날짜 → 덮어쓰기 확인

### 비즈니스 룰
- 1~4: 즉시 차단 (에러 표시)
- 5: 경고 + 덮어쓰기 선택 가능 (M2 WARN_DUP)
- 덮어쓰기 시: `overwrite: true` payload 포함
