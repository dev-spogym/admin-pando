submitBodyComp 함수:
async function submitBodyComp(overwrite = false) {
  setIsSubmitting(true)
  try {
    const payload = {
      member_id: selectedMemberId,
      branch_id: branchId,
      measured_at: measureDate,
      weight: Number(formValues.weight),
      muscle_mass: formValues.muscleMass ? Number(formValues.muscleMass) : null,
      body_fat_percentage: formValues.bodyFatPercentage ? Number(formValues.bodyFatPercentage) : null,
      bmi: formValues.bmi ? Number(formValues.bmi) : null,
      body_fat_mass: formValues.bodyFatMass ? Number(formValues.bodyFatMass) : null,
      basal_metabolic_rate: formValues.basalMetabolicRate ? Number(formValues.basalMetabolicRate) : null,
      source: 'manual',
      review_status: 'auto_approved',  // 수기 입력은 자동 검수완료
      memo,
      created_by: userId,
      overwrite,
    }

    let result
    if (overwrite) {
      // 기존 데이터 업데이트
      const { data, error } = await supabase
        .from('bodyComposition')
        .update(payload)
        .eq('member_id', selectedMemberId)
        .eq('measured_at', measureDate)
        .select()
        .single()
      if (error) throw error
      result = data
    } else {
      const { data, error } = await supabase
        .from('bodyComposition')
        .insert(payload)
        .select()
        .single()
      if (error?.code === '23505') {
        setDuplicateDate(true)  // 중복 재감지 → 덮어쓰기 확인
        return
      }
      if (error) throw error
      result = data
    }

    // 성공
    toast.success('체성분이 등록되었습니다')
    onSuccess(result)
    onClose()
  } catch (e) {
    toast.error('등록에 실패했습니다')
  } finally {
    setIsSubmitting(false)
  }
}

## 📝 디스크립션

### API 응답 분기 (M3 플로우)
| 응답 | 처리 |
|------|------|
| 200 OK | 성공 토스트 + 모달 닫힘 + 부모 갱신 |
| 409/23505 중복 | duplicateDate=true → 덮어쓰기 확인 |
| 422 유효성 | toast.error('입력값을 확인해 주세요') |
| 500 | toast.error('등록 실패') + 폼 유지 |

### 부모 컴포넌트 갱신
| 부모 | 갱신 내용 |
|------|-----------|
| SCR-I006 | 목록 최상단 새 행 추가 (source='manual', review_status='auto_approved') |
| SCR-I006 KPI | 이번 달 측정 건수 +1 / 수기 +1 |
| SCR-I007 | 체성분 카드: 미측정 → 최신 수치 표시 |

### 비즈니스 룰
- 수기 입력: `review_status = 'auto_approved'` (검수 불필요)
- InBody/파일 파싱: `review_status = 'pending'` (검수 필요)
- 덮어쓰기: 기존 레코드 UPDATE (INSERT 아님)
- 저장 성공 후 폼 초기화 (연속 등록 대비)

### 에지 케이스
- 422 유효성 오류: 서버 측 추가 검증 실패 → 입력 재확인 안내
- 타임아웃 10초: AbortController 적용
- 덮어쓰기 성공: "체성분이 업데이트되었습니다" 토스트 (별도 메시지)
