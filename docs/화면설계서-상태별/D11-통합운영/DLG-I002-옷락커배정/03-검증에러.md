검증 로직:
const handleAssign = async () => {
  const newErrors: { locker?: string; status?: string } = {}

  // 1. 락커 선택 여부
  if (!selectedLocker) {
    newErrors.locker = '배정할 락커를 선택해 주세요'
  }

  // 2. 락커 상태 재확인 (선택 후 상태 변경 가능성)
  if (selectedLocker) {
    const { data: freshLocker } = await supabase
      .from('lockers')
      .select('status')
      .eq('id', selectedLocker.id)
      .single()

    if (freshLocker?.status === 'occupied') {
      newErrors.status = '이미 사용 중인 락커입니다. 다른 락커를 선택해 주세요.'
      setSelectedLocker(null) // 선택 초기화
    }
    if (freshLocker?.status === 'maintenance') {
      newErrors.status = '점검 중인 락커입니다. 다른 락커를 선택해 주세요.'
      setSelectedLocker(null)
    }
  }

  if (Object.keys(newErrors).length > 0) {
    setErrors(newErrors)
    return
  }
  await submitAssign()
}

## 📝 디스크립션

### 사용 시점 (Trigger)
- "배정 확인" 클릭 시 검증 실패 (M2 플로우)

### 검증 순서
1. 락커 미선택 → 에러 + 그리드 강조
2. 선택한 락커 상태 재확인 (API) → 사용중/점검중 에러
3. 회원 출석 확인 → 경고 (강제 진행 가능)

### 비즈니스 룰
