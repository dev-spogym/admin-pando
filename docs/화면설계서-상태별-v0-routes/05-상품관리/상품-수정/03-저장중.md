```
Next.js 15 App Router + TypeScript + Tailwind + Supabase 기반 'use client' 컴포넌트를 작성하라.
화면: SCR-P003 상품 수정 — 상태: UPDATE 요청 진행 중

파일: src/app/products/edit/page.tsx

저장 로직:
const handleSubmit = async () => {
  const errors = validateForm(formData);
  if (errors.length > 0) { setErrors(errors); return; }

  setIsSaving(true);
  const { error } = await supabase
    .from('products')
    .update({
      name: formData.name.trim(),
      productType: formData.productType,
      category: formData.category,
      sportType: formData.sportType || null,
      price: formData.price,
      cashPrice: formData.cashPrice || formData.price,
      cardPrice: formData.cardPrice || null,
      duration: formData.duration || null,
      sessions: formData.sessions || null,
      totalCount: formData.sessions || null,
      description: formData.description || null,
      tag: formData.tag || null,
      isActive: formData.isActive,
      kioskVisible: formData.kioskVisible,
      deductionType: formData.deductionType || null,
      holdingEnabled: formData.holdingEnabled,
      transferEnabled: formData.transferEnabled,
      pointAccrual: formData.pointAccrual,
      salesChannel: formData.salesChannel || 'ALL',
      usage_restrictions: formData.usage_restrictions,
      productGroupId: formData.productGroupId || null,
      suspendLimit: formData.suspendLimit || null,
      dailyUseLimit: formData.dailyUseLimit || null,
    })
    .eq('id', productId);

  if (error) {
    toast.error('상품 수정에 실패했습니다. 다시 시도해주세요.');
    setIsSaving(false);
    return;
  }

  toast.success(`'${formData.name}' 상품이 수정되었습니다.`);
  setIsDirty(false);
  moveToPage('/products');
};

UI (isSaving=true):
- 저장 버튼: <Loader2 className="animate-spin" /> "저장 중..." disabled
- 삭제 버튼: disabled
- 취소 버튼: disabled
- 폼 전체: pointer-events-none opacity-60

사용 유틸:
- supabase, moveToPage, toast
- lucide-react: Loader2
```
