// searchParams에서 productId 추출
const searchParams = useSearchParams();
const productId = searchParams.get('id');

useEffect(() => {
  const loadProduct = async () => {
    if (!productId) { moveToPage('/products'); return; }
    setIsLoading(true);
    const [{ data: product }, { data: groups }] = await Promise.all([
      supabase.from('products').select('*').eq('id', productId).single(),
      supabase.from('product_groups').select('*').eq('branchId', branchId)
    ]);
    if (!product) { toast.error('상품을 찾을 수 없습니다.'); moveToPage('/products'); return; }
    setFormData(product);
    setGroups(groups ?? []);
    setIsLoading(false);
  };
  loadProduct();
}, [productId]);

사용 유틸:
- useSearchParams from 'next/navigation'
- supabase, getBranchId, moveToPage, toast
- lucide-react: Package

## 📝 디스크립션

### 사용 시점 (Trigger)
- 상품 목록에서 특정 상품 행 클릭 → `/products/edit?id={productId}` 진입
- productId가 URL searchParams에 포함되어야 함

### 필수 데이터
| 블록 | 테이블 | 조건 |
|------|--------|------|
| 상품 상세 | `products` | `id = ?` |
| 분류 목록 | `product_groups` | `branchId = ?` |

### 인터랙션 (User Actions)
1. 로딩 중 — 모든 인터랙션 차단
2. 로드 완료 → `02-기본-폼`
3. productId 없음 → `/products` redirect

### 비즈니스 룰
- productId 미존재 시 즉시 목록으로 redirect
- products + product_groups 병렬 로드

### 에지 케이스
- 삭제된 상품 ID 접근 → "상품을 찾을 수 없습니다" toast + redirect
- 타 지점 상품 ID 접근 → branchId 불일치 → 동일 처리

### 접근성 (A11y)
- `aria-busy="true"` on main
- 스켈레톤 `aria-hidden="true"`

### 연결 화면
- 이전: `/products` (목록)
- 다음: `02-기본-폼`
