interface BranchForm {
  name: string;        // 지점명 (2~30자)
  code: string;        // 지점 코드 (영문 대문자 2~5자, 예: SGJ)
  phone: string;       // 대표 연락처 (예: 02-000-0000)
  address: string;     // 주소 (readOnly, 주소검색 버튼으로 입력)
  addressDetail: string;
  managerId: string;   // 지점 관리자 Select
  openTime: string;    // 운영 시작 시간 (기본: '06:00')
  closeTime: string;   // 운영 종료 시간 (기본: '23:30')

검증 함수:
const isValidPhone = (v: string) => /^(0\d{1,2})-(\d{3,4})-(\d{4})$/.test(v);
const isValidCode = (v: string) => /^[A-Z]{2,5}$/.test(v);
const isValidName = (v: string) => v.trim().length >= 2 && v.trim().length <= 30;

const validateForm = (): boolean => {
  const errors: FormErrors = {};
  if (!isValidName(branchForm.name)) errors.name = '지점명은 2~30자로 입력해주세요.';
  if (!branchForm.code.trim()) errors.code = '지점 코드를 입력해주세요.';
  else if (!isValidCode(branchForm.code)) errors.code = '지점 코드는 영문 대문자 2~5자여야 합니다. (예: SGJ)';
  else if (branches.some(b => b.code === branchForm.code)) errors.code = '이미 사용 중인 지점 코드입니다.';
  if (!branchForm.phone.trim()) errors.phone = '연락처를 입력해주세요.';
  else if (!isValidPhone(branchForm.phone)) errors.phone = '올바른 전화번호 형식으로 입력해주세요. (예: 02-000-0000)';
  if (!branchForm.address.trim()) errors.address = '주소를 입력해주세요.';
  if (!branchForm.managerId) errors.managerId = '지점 관리자를 선택해주세요.';
  setFormErrors(errors);
  return Object.keys(errors).length === 0;
};

코드 자동생성:
const handleAutoGenerateCode = () => {
  const nextNum = String(branches.length + 1).padStart(3, '0');
  const generated = `SG${nextNum}`;
  setBranchForm(f => ({ ...f, code: generated }));
  setCodeAutoGenerated(true);
  setFormErrors(prev => ({ ...prev, code: undefined }));
};

등록 핸들러:
const handleAddBranchSubmit = async () => {
  if (!validateForm()) return;
  if (isSaving) return;
  setIsSaving(true);
  try {
    const { data, error } = await supabase.from('branches').insert({
      name: branchForm.name,
      code: branchForm.code,
      phone: branchForm.phone,
      address: branchForm.address + (branchForm.addressDetail ? ` ${branchForm.addressDetail}` : ''),
      status: 'active',
      regDate: new Date().toISOString().slice(0, 10),
    }).select().single();
    if (error) { toast.error(`지점 등록 실패: ${error.message}`); return; }
    toast.success(`지점 "${branchForm.name}"이(가) 등록되었습니다.`);
    setBranches(prev => [...prev, data]);
    setIsAddBranchOpen(false);
    setBranchForm(EMPTY_FORM);
    setFormErrors({});
  } finally {
    setIsSaving(false);
  }
};

## 📝 디스크립션

### 사용 시점 (Trigger)
- PageHeader "신규 지점 등록" 버튼 클릭 → isAddBranchOpen=true
- `03-빈상태`의 CTA "신규 지점 등록" 버튼 클릭

### 필수 데이터
| 항목 | 설명 |
|------|------|
| branchForm | EMPTY_FORM으로 초기화 |
| formErrors | 필드별 에러 메시지 |
| branches | 코드 중복 검사에 사용 |
| isSaving | 중복 제출 방지 |

### 인터랙션 (User Actions)
1. 지점 코드 "자동생성" → `SG${branches.length+1}` (3자리 0패딩)
2. 주소 "주소검색" → 주소 자동 입력 (readOnly 필드)
3. 필드 변경 → 해당 필드 에러 즉시 초기화
4. "등록 완료" → validateForm() → supabase insert → toast.success → 모달 닫기
5. "취소" / X 버튼 → handleCloseAddBranch() → 폼 초기화

### 비즈니스 룰
- 필수 필드: 지점명 / 지점코드 / 연락처 / 주소 / 지점관리자
- 지점코드: 영문 대문자 2~5자, 중복 불가 (branches 배열에서 실시간 체크)
- 등록 후 status: 'active', regDate: 오늘 날짜 자동 설정
- isSaving=true 중 중복 클릭 방지

### 에지 케이스
- 코드 중복: "이미 사용 중인 지점 코드입니다." 에러 표시
- supabase insert 실패 → toast.error(`지점 등록 실패: ${error.message}`)
- 모달 닫기 시 EMPTY_FORM 초기화 + formErrors 초기화

### 접근성 (A11y)
- 모달: `role="dialog"` `aria-modal="true"` `aria-labelledby="modal-title"`
- 필수 필드: `aria-required="true"`
- 에러 메시지: `aria-describedby` 연결
