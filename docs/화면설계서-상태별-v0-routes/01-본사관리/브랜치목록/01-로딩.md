useEffect(() => {
  const fetchBranches = async () => {
    const { data, error } = await supabase.from('branches').select('*');
    if (!error && data) {
      setBranches(data);
      // 지점별 회원 수, 매출 병렬 집계
      const stats = await Promise.all(data.map(async (b) => {
        const { count: memberCount } = await supabase
          .from('members').select('*', { count: 'exact', head: true }).eq('branchId', b.id);
        const { data: salesData } = await supabase
          .from('sales').select('amount').eq('branchId', b.id);
        const totalSales = salesData?.reduce((sum, s) => sum + Number(s.amount || 0), 0) ?? 0;
        return { branch: b.name, members: memberCount ?? 0, sales: formatNumber(totalSales), attendance: '-' };
      }));
      setIntegratedStats(stats);
    }
  };
  fetchBranches();
}, []);

## 📝 디스크립션

### 사용 시점 (Trigger)
- `/branches` 첫 진입 시 branches 테이블 + 회원/매출 병렬 집계 로딩 중

### 필수 데이터
| 항목 | 초기값 | 설명 |
|------|--------|------|
| branches | [] | supabase branches 테이블 |
| integratedStats | [] | 지점별 회원/매출 집계 결과 |
| activeTab | 'list' | 현재 탭 키 |

### 인터랙션 (User Actions)
1. 로딩 중: PageHeader 버튼 disabled
2. TabNav 탭 클릭 → 탭 전환 (로딩 중에도 허용)

### 비즈니스 룰
- branches 패치 완료 후 각 지점별 memberCount + salesData 병렬 집계 (Promise.all)
- 집계 완료 → `02-기본-목록` 전환
- branches.length === 0 → `03-빈상태` 전환

### 에지 케이스
- supabase 쿼리 실패 → toast.error 표시 + 빈 상태 유지
- 지점 수 많을수록 Promise.all 소요 시간 증가

### 접근성 (A11y)
- 스피너: `aria-label="지점 목록 불러오는 중"` `role="status"`
- 버튼 disabled: `aria-disabled="true"`

### 연결 화면
- 이전: 사이드바 메뉴
- 다음: `02-기본-목록` (성공) / `03-빈상태` (지점 없음)
