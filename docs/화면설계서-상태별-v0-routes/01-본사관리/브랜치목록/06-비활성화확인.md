const openStatusChange = async (row: any, type: 'closed' | 'inactive') => {
  setStatusChangeTarget(row);
  setStatusChangeType(type);
  if (type === 'closed') {
    const { count: activeMembers } = await supabase.from('members')
      .select('*', { count: 'exact', head: true }).eq('branchId', row.id).eq('status', 'ACTIVE');
    const { count: activeStaff } = await supabase.from('staff')
      .select('*', { count: 'exact', head: true }).eq('branchId', row.id).eq('isActive', true);
    setStatusChangeInfo({ activeMembers: activeMembers ?? 0, activeStaff: activeStaff ?? 0 });
  } else {
    setStatusChangeInfo(null);
  }
  setIsStatusChangeOpen(true);
};

const handleStatusChangeConfirm = async () => {
  if (!statusChangeTarget || !statusChangeType) return;
  if (isStatusChangeSaving) return;
  setIsStatusChangeSaving(true);
  try {
    const newStatus = statusChangeType === 'closed' ? 'closed' : 'inactive';
    const { error } = await supabase.from('branches')
      .update({ status: newStatus }).eq('id', statusChangeTarget.id);
    if (error) { toast.error(`상태 변경 실패: ${error.message}`); return; }
    setBranches(prev => prev.map(b => b.id === statusChangeTarget.id ? { ...b, status: newStatus } : b));
    const label = statusChangeType === 'closed' ? '폐점' : '임시휴업';
    toast.success(`"${statusChangeTarget.name}" 지점이 ${label} 처리되었습니다.`);
    setIsStatusChangeOpen(false);
  } finally {
    setIsStatusChangeSaving(false);
  }
};

## 📝 디스크립션

### 사용 시점 (Trigger)
- DataTable actions "비활성" 버튼 클릭 → openDeactivate(row) → isDeactivateOpen=true (Step 1)
- DataTable actions "상태변경" Select → 임시휴업/폐점 선택 → isStatusChangeOpen=true

### 필수 데이터
| 항목 | 설명 |
|------|------|
| deactivateTarget | 비활성화 대상 지점 row |
| deactivateStep | 1: 영향 범위 / 2: 최종 확인 |
| statusChangeInfo | 폐점 시 활성회원/직원 수 |

### 인터랙션 (User Actions)
1. Step 1 "계속 진행" → deactivateStep=2 (2단계로 이동)
2. Step 2 "이전" → deactivateStep=1 (1단계로 복귀)
3. Step 2 "최종 비활성화" → toast.success + 모달 닫기
4. "취소" / X → isDeactivateOpen=false
5. "상태변경" Select (임시휴업) → 단순 확인 다이얼로그
6. "상태변경" Select (폐점) → 활성회원/직원 수 조회 후 경고 다이얼로그

### 비즈니스 룰
- 비활성화: 2단계 프로세스 (영향 범위 확인 → 최종 확인)
- 진행중 계약 있을 경우 Step 1에서 경고 텍스트 추가 표시
- 폐점 처리: supabase에서 활성 회원/직원 수 실시간 조회 후 표시
- 임시휴업: 단순 확인 → supabase update status='inactive'
- 폐점: 경고 확인 → supabase update status='closed'

### 에지 케이스
- activeContracts=0: 계약 관련 경고 텍스트 미표시
- supabase 상태 변경 실패 → toast.error + 모달 유지
- isStatusChangeSaving=true 중 "확인" 재클릭 방지

### 접근성 (A11y)
- 모달: `role="alertdialog"` (파괴적 작업 경고)
- 위험 버튼 variant="danger": 충분한 색상 대비 + `aria-label`
- Step 진행 상태: 시각적으로 구분 (1단계 배경색 / 2단계 아이콘)
