const fetchAll = useCallback(async () => {
  try {
    const tenantId = user?.tenantId ? Number(user.tenantId) : undefined;
    const [branchRes, auditRes] = await Promise.all([
      getBranches(tenantId),
      getAuditLogs({ page: 1, size: 10 }),
    ]);
    if (branchRes.success) {
      setBranches(branchRes.data);
      await fetchBranchStats(branchRes.data); // 지점별 5쿼리 병렬
    } else {
      toast.error('지점 목록을 불러오지 못했습니다.');

    setLastRefreshed(new Date());

    toast.error('대시보드 데이터를 불러오지 못했습니다.');
  } finally {
    setLoading(false);
  }
}, [user?.tenantId, fetchBranchStats]);

## 📝 디스크립션

### 사용 시점 (Trigger)
- `/super-dashboard` 첫 진입
- 새로고침 버튼 클릭 (isRefreshing=true)

### 필수 데이터
| 항목 | 초기값 | 설명 |
|------|--------|------|
| loading | true | 전체 로딩 플래그 |
| isSuperAdmin | authStore | false → 권한없음 UI |
| branches | [] | 지점 목록 |
| kpiData | 모든 값 0 | 전체 KPI |

### 인터랙션 (User Actions)
1. 권한 없음 → "대시보드로 이동" → router.push('/')
2. 새로고침 버튼: isRefreshing=true 시 animate-spin

### 비즈니스 룰
- isSuperAdmin=false → 권한없음 UI 즉시 렌더 (fetchAll 호출 안 함)
- fetchAll: getBranches + getAuditLogs 병렬 → fetchBranchStats 순차

### 에지 케이스
- getBranches 실패 → toast.error + loading=false
- tenantId null → undefined → 전체 지점 조회

### 접근성 (A11y)
- 로딩 스켈레톤: `aria-busy="true"`
- 새로고침: `aria-label="대시보드 새로고침"`
