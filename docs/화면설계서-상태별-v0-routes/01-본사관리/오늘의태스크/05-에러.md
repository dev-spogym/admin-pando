useEffect(() => {
  try {
    const saved = localStorage.getItem(STORAGE_KEY);
    if (!saved) return;
    const parsed = JSON.parse(saved) as EditableTask[];
    if (Array.isArray(parsed)) setCustomTasks(parsed);
  } catch {
    localStorage.removeItem(STORAGE_KEY); // 손상된 데이터 제거
    // customTasks = [] 유지 (에러 표시 없음)
  }
}, []);

에러 표시 미구현 (현재):

## 📝 디스크립션

### 사용 시점 (Trigger)
- localStorage JSON 파싱 실패 (손상된 데이터)
- getTodayTasks() useMemo 런타임 오류 (Next.js Error Boundary)
- isSuperAdmin 폼 제출 시 title 미입력 (조용한 차단)

### 필수 데이터
| 항목 | 조건 | 설명 |
|------|------|------|
| localStorage | 파싱 실패 | catch → removeItem → 복구 |
| randomTasks | useMemo 에러 | React Error Boundary 처리 |

### 인터랙션 (User Actions)
1. localStorage 에러: 자동 복구 (사용자 인지 불가)
2. Error Boundary (향후): "다시 시도" 버튼 → reset()
3. 폼 제출 에러: title 미입력 → 버튼 클릭 무반응

### 비즈니스 룰
- localStorage 실패: 무음 복구 (customTasks=[], 에러 표시 없음)
- randomTasks: 동기 함수이므로 런타임 에러 거의 없음
- 폼 유효성: title 미입력만 체크 (에러 메시지 미표시)
- 현재 구현상 명시적 에러 상태 없음 — 에러는 최소화

### 에지 케이스
- authUser=null (인증 만료): role='MANAGER' 기본값 → randomTasks 정상 표시
- STORAGE_KEY 데이터 타입 불일치: Array.isArray 체크로 방어
- 브라우저 localStorage 비활성화: catch → customTasks=[] 유지

### 접근성 (A11y)
- 폼 유효성 에러 (향후): `aria-invalid="true"` `aria-describedby` 연결
- Error Boundary 버튼: `aria-label="오늘의 태스크 다시 시도"`
