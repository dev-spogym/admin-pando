  // 스캔 타임아웃
  useEffect(() => {
    if (!isScanning) return;
    const timeout = setTimeout(() => {
      setIsScanning(false);
      setScanError('timeout');
    }, 30000);
    return () => clearTimeout(timeout);
  }, [isScanning]);

  // 데이터 로드 에러
  const fetchCards = async () => {
    try {
      setIsLoading(true);
      const { data, error } = await supabase.from('rfid_cards').select('...');
      if (error) throw error;
      setCards(data);
    } catch (e) {
      setLoadError(true);
      toast.error("카드 목록을 불러오지 못했습니다.");
    } finally {
      setIsLoading(false);
    }
  };

데이터:
  - scanError: 'timeout' | 'duplicate' | null
  - loadError: boolean

## 📝 디스크립션

### 사용 시점 (Trigger)
- 스캔 30초 타임아웃 → `scanError = 'timeout'`
- 중복 카드번호 감지 → `scanError = 'duplicate'`
- 데이터 fetch 실패 → `loadError = true`

### 필수 데이터
- 에러 유형 상태값 (`scanError`, `loadError`)

### 인터랙션 (User Actions)
1. [다시 시도] → 스캔 재시작 또는 데이터 재조회
2. [취소] → 에러 상태 초기화

### 비즈니스 룰
- 스캔 타임아웃 30초 (configurable)
- 중복 카드번호는 insert 전 select로 사전 확인

### 에지 케이스
- 연속 3회 스캔 실패 → "리더기를 확인하세요." 추가 안내

### 접근성 (A11y)
- 에러 영역 `role="alert"` (즉시 읽기)
- `aria-live="assertive"` for 에러 메시지
