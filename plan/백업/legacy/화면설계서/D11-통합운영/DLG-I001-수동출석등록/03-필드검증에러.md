interface FormErrors {
  member?: string
  channel?: string
  time?: string
  membership?: string
}

const handleSubmit = async () => {
  const newErrors: FormErrors = {}

  // 1. 회원 선택 검증
  if (!selectedMember) {
    newErrors.member = '회원을 선택해 주세요'
    memberSearchRef.current?.focus()
  }

  // 2. 채널 선택 검증
  if (!selectedChannel) {
    newErrors.channel = '출석 채널을 선택해 주세요'
  }

  // 3. 시간 유효성 검증
  const selectedTime = new Date(attendanceTime)
  if (selectedTime > new Date()) {
    newErrors.time = '미래 시간은 입력할 수 없습니다'
  }

  // 4. 회원권 유효성 검증 (선택 회원의 계약 상태 확인)
  if (selectedMember) {
    const { data: contract } = await supabase
      .from('contracts')
      .select('status, expires_at')
      .eq('member_id', selectedMember.id)
      .eq('status', '서명완료')
      .single()
    if (!contract || new Date(contract.expires_at) < new Date()) {
      newErrors.membership = '회원권이 만료되었거나 유효하지 않습니다. 강제 등록하려면 아래 버튼을 선택하세요.'
    }
  }

  if (Object.keys(newErrors).length > 0) {
    setErrors(newErrors)
    return
  }
  await submitAttendance()
}

## 📝 디스크립션

### 사용 시점 (Trigger)
- 저장 버튼 클릭 시 검증 실패

### 검증 순서 (M2 플로우 기준)
1. 회원 선택 여부 → 에러 시 검색 필드 포커스
2. 채널 선택 여부
3. 시간 유효성 (미래 시간 불가)
4. 회원권 유효성 (만료/정지) → 경고 + 강제 등록 옵션

### 인터랙션 (User Actions)
1. 에러 필드 수정 → 해당 에러 제거
2. 강제 등록 → forceRegister=true로 API 호출
3. 강제 등록 취소 → 에러 상태 유지

### 비즈니스 룰
- 에러 1~3: 즉시 차단 (저장 불가)
- 에러 4 (회원권 만료): 경고 표시 + 강제 등록 선택 가능
- 강제 등록 시: API payload에 `force: true` 포함

### 에지 케이스
- 여러 에러 동시: 모든 에러 동시 표시
- 강제 등록 중 API 실패: `05-API에러` 처리
