const [isSaving, setIsSaving] = useState(false)

handleSave:
async function handleSave() {
  setIsSaving(true)
  try {
    await supabase.from('kiosk_settings').upsert({ branch_id, ...formValues })
    toast.success('설정이 저장되었습니다')
    setIsDirty(false)
  } catch (e) {
    toast.error('저장에 실패했습니다')
  } finally {
    setIsSaving(false)
  }
}

[저장 중 버튼 상태]
<button
  disabled={isSaving}
  className="px-6 py-2.5 bg-blue-600 text-white text-sm font-medium rounded-lg
             disabled:opacity-60 disabled:cursor-not-allowed flex items-center gap-2"
>
  {isSaving ? (
    <>
      <svg className="animate-spin w-4 h-4" /> 저장 중...
    </>
  ) : '설정 저장'}
</button>

- 폼 전체 pointer-events-none (isSaving 중)
- 모든 입력 필드 disabled
- 저장 완료: isDirty = false → 버튼 다시 비활성화
