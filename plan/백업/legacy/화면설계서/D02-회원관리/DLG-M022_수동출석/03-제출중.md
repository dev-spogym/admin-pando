3. **다른 버튼은 정상**: 수정/프로필 등 주변 버튼은 계속 동작
4. **페이지 이동 허용**: 내비게이션 가능 (요청은 백그라운드 진행)

    const ac = new AbortController();
    const t = setTimeout(() => ac.abort(), 5_000);
    return fetch('/api/attendances', {
      method:'POST',
      signal: ac.signal,
      headers:{'Content-Type':'application/json'},
      body: JSON.stringify({ memberId, type:'MANUAL', checkInMethod:'MANUAL', checkInAt: new Date().toISOString() }),
    }).finally(() => clearTimeout(t))
      .then(r => r.ok ? r.json() : r.json().then(Promise.reject.bind(Promise)));
  },
  ...
});
