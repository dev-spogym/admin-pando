const handleDelete = async () => {
  const { error } = await supabase.from('today_tasks').delete().eq('id', task.id)
  if (!error) {
    toast.success("태스크가 삭제되었습니다.") // Toast_Del
    onClose()
    onTaskDeleted(task.id) // 목록에서 제거

  }
}

[완료 처리 — E_D98_02_M1_10 / E_D98_02_M3_09]
const handleComplete = async () => {
  const { error } = await supabase
    .from('today_tasks')
    .update({ status: 'completed', completed_at: new Date().toISOString() })
    .eq('id', task.id)

  if (!error) {
    const allDone = tasks.every(t => t.id === task.id || t.status === 'completed')
    if (allDone) {
      // E_D98_02_M3_11: 전체 완료
      toast.success("오늘 할일을 모두 완료했습니다!") // AllDone

      toast.success("완료 처리되었습니다.") // Toast_Complete
    }
    onClose()

  }
}
