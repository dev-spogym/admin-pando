## 비즈니스 룰 델타
- 서버 가드 우선 (middleware.ts 403/redirect)
- SSR off 환경에서 클라 fallback 컴포넌트 렌더
- audit_log `PERMISSION_DENIED` 기록 (1분 throttle)
- 사이드바 `getMenuForRole` 제외 처리
- owner가 타 브랜드 URL 조작 → 이 상태 대신 `E403003` toast + `/` redirect

## 에지 케이스
- 역할 변경 직후 캐시 토큰: 강제 logout 또는 refresh
- primary → owner 강등 직후: 전 지점 → 자기 브랜드만, 타 브랜드 URL 접근 시 `E403003`
- JWT 만료와 동시: `/login?redirect=/branches` 우선
- SEO 봇: robots.txt disallow + 403

## 바이브코딩 프롬프트 (델타)
```
// middleware.ts
export async function middleware(req) {
  const user = await getSessionUser(req);
  if (req.nextUrl.pathname.startsWith('/branches')) {
    if (!user) return NextResponse.redirect('/login?redirect=/branches');
    if (!['primary','superAdmin','owner'].includes(user.role)) {
      logAudit({ action:'PERMISSION_DENIED', userId:user.id, targetType:'page', targetId:'SCR-092' });
      return NextResponse.redirect('/forbidden?from=/branches');
    }
  }
  return NextResponse.next();
}

// 클라이언트 fallback
export default function BranchesPage() {
  const { user } = useAuthStore();
  const canAccess = user && ['primary','superAdmin','owner'].includes(user.role);
  if (!canAccess) return <ForbiddenCard user={user} requiredRoles={['primary','superAdmin','owner']}/>;
  return <BranchManagement/>;
}

function ForbiddenCard({ user, requiredRoles }) {
  return (
    <main className="min-h-screen flex items-center justify-center bg-gray-50 px-4">
      <div className="max-w-md w-full bg-white rounded-2xl shadow-sm ring-1 ring-gray-100 p-8 text-center">
        <div className="mx-auto h-16 w-16 rounded-full bg-red-100 flex items-center justify-center mb-4">
          <ShieldAlert className="h-8 w-8 text-red-600"/>
        </div>
        <h1 className="text-xl font-bold text-gray-900">접근 권한이 없습니다</h1>
        <p className="mt-3 text-sm text-gray-600">지점 관리 페이지는 최고관리자 또는 센터장만 접근할 수 있어요.</p>
        <div className="mt-4 rounded-lg bg-gray-50 p-3 text-xs text-gray-600">
          <p>현재 역할: <span className="font-mono">{user?.role ?? 'unknown'}</span></p>
          <p>필요 역할: <span className="font-mono">{requiredRoles.join(' · ')}</span></p>
        </div>
        <div className="mt-6 flex gap-2 justify-center">
          <Button variant="primary" onClick={() => router.push('/')}>← 대시보드로 이동</Button>
          <Button variant="secondary" onClick={() => window.open('/help/roles')}>도움말 보기</Button>
        </div>
      </div>
    </main>
  );
}
```
