## 🧩 바이브코딩 프롬프트

```
모달: DLG-085-001 공지 등록 — 빈 폼 (OpenNew)

<Modal open={showNoticeModal}>
  <ModalContent className="max-w-2xl">
    <ModalHeader>
      <ModalTitle>{editingNotice ? '공지 수정' : '공지 등록'}</ModalTitle>
    </ModalHeader>
    <ModalBody className="space-y-4">
      <!-- 제목 (필수) M2_085001_01~02 -->
      <div>
        <label className="text-sm font-medium">제목 <span className="text-red-500">*</span></label>
        <Input
          value={form.title}
          onChange={e => setForm(p => ({...p, title: e.target.value}))}
          placeholder="공지 제목을 입력하세요"
          className="mt-1"
        />
        {errors.title && <p className="text-xs text-red-500 mt-1">{errors.title}</p>}
      </div>
      <!-- 내용 (필수) M2_085001_03~04 -->
      <div>
        <label className="text-sm font-medium">내용 <span className="text-red-500">*</span></label>
        <Textarea
          value={form.content}
          onChange={e => setForm(p => ({...p, content: e.target.value}))}
          rows={6}
          placeholder="공지 내용을 입력하세요"
          className="mt-1"
        />
        {errors.content && <p className="text-xs text-red-500 mt-1">{errors.content}</p>}
      </div>
      <!-- 옵션 행 -->
      <div className="flex items-center gap-6">
        <label className="flex items-center gap-2 text-sm">
          <Toggle checked={form.isPinned} onChange={v => setForm(p => ({...p, isPinned: v}))} />
          상단 고정
        </label>
        <label className="flex items-center gap-2 text-sm">
          <Toggle checked={form.sendNotification} onChange={v => setForm(p => ({...p, sendNotification: v}))} />
          알림 발송
        </label>
      </div>
      <!-- 첨부파일 M2_085001_05~09 -->
      <div>
        <label className="text-sm font-medium">첨부파일</label>
        <input
          type="file"
          accept=".pdf,.doc,.docx,.jpg,.png"
          onChange={handleFileChange}
          className="mt-1 text-sm"
        />
        {errors.file && <p className="text-xs text-red-500 mt-1">{errors.file}</p>}
      </div>
    </ModalBody>
    <ModalFooter className="flex justify-end gap-2">
      <Button variant="outline" onClick={handleClose}>취소</Button>
      <Button
        disabled={!form.title || !form.content || !!errors.file}
        onClick={handleSave}
      >저장</Button>
    </ModalFooter>
  </ModalContent>
</Modal>
```
