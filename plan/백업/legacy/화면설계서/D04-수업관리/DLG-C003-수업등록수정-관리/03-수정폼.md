useEffect(() => {
  if (lesson && mode === 'edit') {
    reset({
      title: lesson.class.name,
      instructorId: lesson.instructor_id,
      type: lesson.class.type,
      startTime: format(new Date(lesson.start_at), 'HH:mm'),
      endTime: format(new Date(lesson.end_at), 'HH:mm'),
      capacity: lesson.capacity,
      memo: lesson.memo || '',
    })
  }
}, [lesson, mode])

변경 감지 (isDirty):
- 변경 없으면 닫기 시 바로 닫힘
- 변경 있으면 `04-미저장확인` 다이얼로그
